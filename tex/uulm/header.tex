\newcommand{\chapter}[2][]{
	\addtocounter{framenumber}{-1}
	\section{#2}
	\renewcommand{\thesubchapter}{}
		
	\begin{frame}[plain]
		\begin{columns}
			\begin{column}{0.75\linewidth}
				\vskip4cm
				\begin{mdframed}[topline = false, rightline = false, bottomline = false, linewidth = 4pt, linecolor = uulmblue]			
					{\Huge \color{uulmblue} \textbf{\insertsection\phantom{/Üyg}}}
				\end{mdframed}		
				\vskip2ex
				{\large \color{uulmblue} \textbf{\phantom{/Ü,\glqq yg\grqq}}}
			\end{column}
			\hfill
			\begin{column}{0.2\linewidth}
					\vskip3cm
					\ifthenelse{\equal{#1}{}}{}{
						\begin{figure}
							\begin{center}					
								\includegraphics[width=0.8\linewidth]{#1}
							\end{center}
						\end{figure}
					}
			\end{column}
		\end{columns}	
	\end{frame}

}
